@Override
protected boolean preCommit() {
    if(super.preCommit()){
        try {
            if (PersistenceHelper.isNew(getItem())) {
                changeQueryService.createControlChange(getItem(), ChangeQueryType.CREATE, "Заявка отправлена");
            } else {
                changeQueryService.createControlChange(getItem(), ChangeQueryType.UPDATE, "Заявка отправлена");
            }
        } catch (ChangeQueryException e) {
            e.printStackTrace();
        }
    }
    return true;
}
