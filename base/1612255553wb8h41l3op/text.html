select string_to_array('Иванов Иван Иваныч', ' ') @> string_to_array('Иван Иваныч Иванов', ' ');


--Сравнивает по наименованию

with shares as (

select

distinct cre.id::uuid as id,

round((split_part(share, '/', 1)::bigint)/(split_part(share, '/', 2)::bigint)::numeric, 5) as sh

from rtneo_real_estate re join t_dfgv98 t on re.cadastral_number = t.cad

join rtneo_contragent_real_estate cre on cre.real_estate_id = re.id

join rtneo_contragent c on cre.contragent_id = c.id

where

cre.share_ is null

and (not t.share = ''

and t.share is not null)

and t.share similar to '[0-9]*/[0-9]*'

and string_to_array(t.fio_data, ' ') @> string_to_array(c.name, ' ')

)

update

rtneo_contragent_real_estate cre

set

share_ = shares.sh

from

shares

where

cre.id = shares.id;
