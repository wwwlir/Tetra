BEGIN WORK;


with shares as (

select

"cre_id"::uuid as id,

"cad" as cad,

round((split_part(share, '/', 1)::bigint)/(split_part(share, '/', 2)::bigint)::numeric, 5) as sh

from

t_dfgv98

where

share SIMILAR TO '[0-9]*/[0-9]*'

order by

"cad")

update

rtneo_contragent_real_estate cre

set

share_ = shares.sh

from

shares

where

cre.id = shares.id


COMMIT WORK;
