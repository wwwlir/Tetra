select distinct r.contragentRealEstate.id, r.contragentRealEstate.realEstate.address, r.contragentRealEstate.realEstate.cadastralNumber, r.contragentRealEstate.realEstate.area, r.contragentRealEstate.category.name, r.contragentRealEstate.calculationAmount, r.containerYard.code, (select sum(cw.count) from rtneo$ContainerWaste cw where cw.containerYard = r.containerYard), (select sum(cw.type.volume*cw.count) from rtneo$ContainerWaste cw where cw.containerYard = r.containerYard), r.containerYard.fixedExportSchedule from rtneo$Contragent c join rtneo$Payment p on c.inn = p.inn join rtneo$RealEstateContainerYard r on c.id = r.contragentRealEstate.contragent.id where c.personalAccount = 'ААА3812014066' order by r.contragentRealEstate.realEstate.cadastralNumber
