// Эта аннотация позволяет внедрить в класс Manager поведение метода из класса Employee 

class Employee {
    def doWork() { 'my work' }
}

class Manager {
    @Delegate
    Employee slave = new Employee()
}

def worker = new Manager()
println worker.doWork()
